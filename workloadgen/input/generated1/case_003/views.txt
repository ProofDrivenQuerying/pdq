SELECT DISTINCT lineitem.l_orderkey
FROM lineitem, orders, part
WHERE lineitem.l_orderkey = orders.o_orderkey
	 AND part.p_retailprice = lineitem.l_extendedprice
	 AND orders.o_orderpriority = '5-LOW'
	 AND lineitem.l_shipinstruct = 'TAKE BACK RETURN'
	 AND lineitem.l_shipmode = 'REG AIR'
	 AND part.p_brand = 'Brand#51'
