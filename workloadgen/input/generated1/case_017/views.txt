SELECT DISTINCT lineitem.l_linenumber
FROM part, lineitem
WHERE part.p_retailprice = lineitem.l_extendedprice
	 AND lineitem.l_returnflag = 'R'
	 AND part.p_mfgr = 'Manufacturer#4'
	 AND lineitem.l_shipmode = 'SHIP'
	 AND lineitem.l_linestatus = 'F'
