SELECT DISTINCT orders.o_orderkey
FROM part, orders
WHERE part.p_retailprice = orders.o_totalprice
	 AND orders.o_orderpriority = '3-MEDIUM'
	 AND part.p_mfgr = 'Manufacturer#3'
	 AND orders.o_orderstatus = 'O'

SELECT DISTINCT orders.o_orderkey
FROM orders, customer, part
WHERE orders.o_custkey = customer.c_custkey
	 AND part.p_retailprice = orders.o_totalprice
	 AND orders.o_orderpriority = '3-MEDIUM'
	 AND customer.c_mktsegment = 'MACHINERY'
	 AND part.p_mfgr = 'Manufacturer#3'
	 AND orders.o_orderstatus = 'O'

SELECT DISTINCT nation.n_nationkey
FROM customer, nation
WHERE customer.c_nationkey = nation.n_nationkey
	 AND customer.c_mktsegment = 'MACHINERY'
	 AND nation.n_name = 'MOZAMBIQUE'
