SELECT DISTINCT orders.o_orderkey
FROM lineitem, orders, customer
WHERE orders.o_orderdate = lineitem.l_shipdate
	 AND orders.o_custkey = customer.c_custkey
	 AND orders.o_orderstatus = 'O'
	 AND customer.c_mktsegment = 'HOUSEHOLD'

FROM orders, customer, part, nation
WHERE orders.o_custkey = customer.c_custkey
	 AND part.p_retailprice = orders.o_totalprice
	 AND customer.c_nationkey = nation.n_nationkey
	 AND orders.o_orderstatus = 'O'
	 AND customer.c_mktsegment = 'HOUSEHOLD'
	 AND nation.n_name = 'JORDAN'
	 AND part.p_mfgr = 'Manufacturer#5'
