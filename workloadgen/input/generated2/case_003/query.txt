SELECT DISTINCT nation.n_nationkey, orders.o_orderkey, part.p_partkey
FROM orders, customer, lineitem, nation, part
WHERE orders.o_custkey = customer.c_custkey
	 AND orders.o_orderdate = lineitem.l_receiptdate
	 AND customer.c_nationkey = nation.n_nationkey
	 AND orders.o_totalprice = part.p_retailprice
	 AND orders.o_orderpriority = '2-HIGH'
	 AND lineitem.l_shipmode = 'SHIP'
	 AND lineitem.l_linestatus = 'F'
	 AND customer.c_mktsegment = 'FURNITURE'
